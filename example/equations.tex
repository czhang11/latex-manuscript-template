% NDVI
\begin{equation}
    NDVI = \frac{NIR-Red} {NIR+Red}
\end{equation}

% NDWI
\begin{equation}
    NDWI = \frac{Green-NIR} {Green+NIR}
\end{equation}

% Kappa
\begin{equation}
    \kappa = \frac{(p_0-p_e)} {(1-p_e)}
\end{equation}

% Recall
\begin{equation}
    Recall = \frac{TP} {TP+FN}
\end{equation}

% F1-score
\begin{equation}
    F1 = 2\times\frac{Precision\times Recall} {Precision+Recall}
\end{equation}

% ReLU
\begin{equation}
    f(z)=max(0,z)
\end{equation}

% SoftMax
\begin{equation}
    \sigma(z)_j = \frac{e^{z_j}} {\sum_{k=1}^{K} e^{z_k}}
\end{equation}
