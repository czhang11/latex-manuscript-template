\begin{figure}[!htb]
    \centering
    % 1st row
    \begin{subfigure}[!htb]{\textwidth}
        \centering
        \includegraphics[width=\textwidth]{img/fig-a}
        \caption{fig-a\\\hspace{\textwidth}}\label{fig:fig-a}
    \end{subfigure}
    % 2nd row
    \begin{subfigure}[!htb]{\textwidth}
        \centering
        \includegraphics[width=\textwidth]{img/fig-b}
        \caption{fig-b\\}\label{fig:fig-b}
    \end{subfigure}
    % Caption
    \caption{Figure}
    \label{fig:fig}
\end{figure}
\begin{figure}[!htb]\ContinuedFloat    
    % 3rd row
    \begin{subfigure}[!htb]{\textwidth}
        \centering
        \includegraphics[width=\textwidth]{img/fig-c}
        \caption{fig-c}\label{fig:fig-c}
    \end{subfigure}
    % Caption
    \renewcommand{\thefigure}{\arabic{figure} (Cont.)}
    \caption{}
    \renewcommand{\thefigure}{\arabic{figure}}
\end{figure}
